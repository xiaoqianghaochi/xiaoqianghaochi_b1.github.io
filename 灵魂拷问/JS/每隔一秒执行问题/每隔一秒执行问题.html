# 每隔一秒执行问题

for (var i=0; i<5; i++) {
  setTimeout(() => console.log(i), 1000)
}

一秒之后输出 5 个 5

改造函数使其每隔一秒输出 i 的值

# let

for (let i=0; i<5; i++) {
  setTimeout(() => console.log(i), 1000*i)
}

# IIFE

for (var i=0; i<5; i++) {
  (function(i){
    setTimeout(() => console.log(i), 1000*i)
  })(i)
}

# 传值

var out = (i) => {
  setTimeout (() => console.log(i), 1000*i)
}
for (var i=0; i<5; i++) {
  out(i)
}

# Promise.all

let arr = []

const out = i => new Promise(resolve => {
  setTimeout(() => {
    console.log(i)
    resolve()
  }, 1000*i)
})

for (var i=0; i<5; i++) {
  arr.push(out(i))
}

Promise.all(arr)

# Async/await

var sleep = () => new Promise (res => setTimeout(res, 1000));
(async () => {
  for (let i=0; i<5; i++) {
    await sleep()
    console.log(i)
  }
})()

注意 iife 前面的分号很重要
